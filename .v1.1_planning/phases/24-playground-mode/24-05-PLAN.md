```csharp
// Assets/Scripts/Playground/PlaygroundRespawnHandler.cs
using System.Collections;
using UnityEngine;

public class PlaygroundRespawnHandler : MonoBehaviour
{
    [SerializeField] private PlayerHPController playerHP;
    [SerializeField] private Transform spawnPoint;
    [SerializeField] private float respawnDelay = 1.5f; // Claude's discretion

    private void Awake()
    {
        if (playerHP == null)
            playerHP = FindObjectOfType<PlayerHPController>();

        playerHP.onDeathOverride = () => StartCoroutine(RespawnAfterDelay());
    }

    private IEnumerator RespawnAfterDelay()
    {
        // Brief pause so the death flash/animation can play
        yield return new WaitForSecondsRealtime(respawnDelay);

        // Teleport to spawn point
        playerHP.transform.position = spawnPoint.position;

        // Restore full HP
        playerHP.RestoreFullHP(); // Use whatever method restores HP — read PlayerHPController for correct method name
        // If no RestoreFullHP exists, set the HP field directly and refresh the HP bar:
        // playerHP.vidaActual = playerHP.vidaMaxima; (or equivalent)
        // playerHP.RefreshHPBar(); (if method exists)
    }

    private void OnDestroy()
    {
        // Clean up the override when scene unloads
        if (playerHP != null)
            playerHP.onDeathOverride = null;
    }
}
```

Read PlayerHPController.cs to find the correct method for restoring HP (may be RestaurarVida, SetHP, or direct field assignment). Use whichever exists.

Respawn delay is 1.5 seconds — enough time for a death flash without feeling punishing.

**PlaygroundProjectileSpawner.cs:**

```csharp
// Assets/Scripts/Playground/PlaygroundProjectileSpawner.cs
using System.Collections;
using UnityEngine;

public class PlaygroundProjectileSpawner : MonoBehaviour
{
    [SerializeField] private GameObject projectilePrefab; // EnemyProjectile prefab that falls downward
    [SerializeField] private float spawnInterval = 3f;
    [SerializeField] private float projectileSpeed = 5f; // Claude's discretion

    private void Start() => StartCoroutine(SpawnLoop());

    private IEnumerator SpawnLoop()
    {
        while (true)
        {
            yield return new WaitForSeconds(spawnInterval);

            // Do not spawn while shop is open (Time.timeScale = 0)
            if (Time.timeScale > 0f)
            {
                GameObject proj = Instantiate(projectilePrefab, transform.position, Quaternion.identity);

                // Set velocity to fall straight down
                Rigidbody2D rb = proj.GetComponent<Rigidbody2D>();
                if (rb != null)
                    rb.linearVelocity = Vector2.down * projectileSpeed;
            }
        }
    }
}
```

Read EnemyProjectile.cs to confirm how velocity is set (may be rb.velocity, rb.linearVelocity, or a speed field on the component). Use the correct approach.

The projectile spawns at a fixed position (transform.position of the ProjectileSpawnPoint GameObject — left side, ~x: -5, y: 3). Always same spot, predictable. Interval is 3 seconds. Speed is 5 units/second (Claude's discretion — adjust if visually too fast/slow).
