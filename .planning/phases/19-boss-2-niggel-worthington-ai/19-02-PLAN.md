```csharp
    private float pulseTimer = 0f;
    private SpriteRenderer sr;

    private Color pulseColor = new Color(1f, 0.5f, 0f); // orange

        if (sr != null)

            boss?.HealCoinVault(GameBalance.Bosses.Niggel.HealingCoinAmount);
            Destroy(gameObject);
            return;
        }
        if (other.CompareTag("Wall")) { Destroy(gameObject); return; }
        // does not call base — no player damage
    }
}
```

**BlackDotBarrier.cs:**
```csharp
public class BlackDotBarrier : MonoBehaviour
{
    private void Awake()
    {
        int barrierLayer = LayerMask.NameToLayer("BossBarrier");
        int playerLayer = LayerMask.NameToLayer("Default");
        int playerProjLayer = LayerMask.NameToLayer("PlayerProjectiles");

        // Player walks through — ignore player layer
        if (barrierLayer >= 0 && playerLayer >= 0)
            Physics2D.IgnoreLayerCollision(barrierLayer, playerLayer, true);

        // Also ignore enemy projectile collisions with barriers
        int enemyProjLayer = LayerMask.NameToLayer("EnemyProjectile");
        if (barrierLayer >= 0 && enemyProjLayer >= 0)
            Physics2D.IgnoreLayerCollision(barrierLayer, enemyProjLayer, true);

        // PlayerProjectiles DO collide — handled by trigger
    }

    private void OnTriggerEnter2D(Collider2D other)
    {
        if (other.gameObject.layer == LayerMask.NameToLayer("PlayerProjectiles"))
        {
            Destroy(other.gameObject); // block and destroy player spell
        }
    }
}
```

IMPORTANT: The "BossBarrier" layer must be created in Unity Editor (Edit > Project Settings > Tags and Layers). Add it as a user layer. BlackDotBarrier GameObjects must be on that layer. The barrier collider should be a trigger (Is Trigger = true) so it doesn't physically block the player — combined with IgnoreLayerCollision, player passes through freely.

**Attack2_RichDash() coroutine — all enrage levels:**
```csharp
private IEnumerator Attack2_RichDash()
{
    isDashing = true;
    Vector2 dashDir = playerTransform != null
        ? ((Vector2)playerTransform.position - (Vector2)transform.position).normalized
        : Vector2.right;

    Vector3 rawTarget = transform.position + (Vector3)(dashDir * GameBalance.Bosses.Niggel.DashDistance);
    rawTarget.x = Mathf.Clamp(rawTarget.x, arenaCenter.x - arenaClampX, arenaCenter.x + arenaClampX);
    rawTarget.y = Mathf.Clamp(rawTarget.y, arenaCenter.y - arenaClampY, arenaCenter.y + arenaClampY);

    float elapsed = 0f;
    Vector3 startPos = transform.position;
    while (elapsed < GameBalance.Bosses.Niggel.DashDuration)
    {
        transform.position = Vector3.Lerp(startPos, rawTarget, elapsed / GameBalance.Bosses.Niggel.DashDuration);
        elapsed += Time.deltaTime;
        yield return null;
    }
    transform.position = rawTarget;
    isDashing = false;

    // Reset dash cooldown based on enrage level
    dashCooldownTimer = enrageLevel >= 3
        ? GameBalance.Bosses.Niggel.Enrage3DashCooldown
        : enrageLevel >= 2
            ? GameBalance.Bosses.Niggel.Enrage2DashCooldown
            : GameBalance.Bosses.Niggel.BaseDashCooldown;
}
```

```csharp
private IEnumerator CoinShieldRoutine()
{
    coinShieldActive = true;
    // Visual: tint self yellow then pulse blue
    Color original = spriteRenderer.color;
    float chargeTime = GameBalance.Bosses.Niggel.ShieldChargeDuration;
    float t = 0f;

    {
        t += Time.deltaTime;
        spriteRenderer.color = Color.Lerp(Color.yellow, Color.cyan, Mathf.PingPong(t * 3f, 1f));
        yield return null;
    }
    spriteRenderer.color = original;

    // Burst
    int count = GameBalance.Bosses.Niggel.ShieldBurstCount;
    for (int i = 0; i < count; i++)
    {
        float angle = (360f / count) * i;
        Vector2 dir = new Vector2(Mathf.Cos(angle * Mathf.Deg2Rad), Mathf.Sin(angle * Mathf.Deg2Rad));
        if (coinBagPrefab != null)

            float a = Mathf.Atan2(dir.y, dir.x) * Mathf.Rad2Deg;

            var sr = proj.GetComponent<SpriteRenderer>();
            if (sr != null) sr.color = new Color(1f, 0.5f, 0f); // orange tint
        }
    }
    coinShieldActive = false;
```

**Update main StartAttackRoutine() — master attack loop:**
```csharp
private IEnumerator AttackLoop()
{
    while (!isDead)
    {
        float interval = enrageLevel >= 2
            ? GameBalance.Bosses.Niggel.Enrage2AttackInterval
            : enrageLevel >= 1
                ? GameBalance.Bosses.Niggel.Enrage1AttackInterval
                : GameBalance.Bosses.Niggel.BaseAttackInterval;

        yield return new WaitForSeconds(interval);
        if (isDead) break;

        // Enrage 3: sometimes do coin shield instead of regular attack
        if (enrageLevel >= 3 && !coinShieldActive && Random.value < 0.35f)
        {
            shieldRoutine = StartCoroutine(CoinShieldRoutine());
            continue;
        }

        // Enrage 2: sometimes fire healing coin
        if (enrageLevel >= 2 && Random.value < 0.3f && healingCoinPrefab != null)
        {
            FireHealingCoin();
            continue;
        }

        // Default: throw coin bag
        yield return StartCoroutine(Attack1_ThrowMoney());
    }
}

private void FireHealingCoin()
{
    if (playerTransform == null || healingCoinPrefab == null) return;
    Vector2 dir = (playerTransform.position - transform.position).normalized;
    float angle = Mathf.Atan2(dir.y, dir.x) * Mathf.Rad2Deg;
    GameObject hc = Instantiate(healingCoinPrefab, transform.position, Quaternion.Euler(0, 0, angle));
    var rb = hc.GetComponent<Rigidbody2D>();
    if (rb != null) rb.velocity = dir * GameBalance.Bosses.Niggel.HealingCoinSpeed;
    var hcp = hc.GetComponent<HealingCoinProjectile>();
    if (hcp != null) hcp.boss = this;
}
```
