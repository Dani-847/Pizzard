```csharp
// Assets/Scripts/Playground/DummyDPSTracker.cs
using System.Collections.Generic;
using System.Linq;
using UnityEngine;
using TMPro;

public class DummyDPSTracker : MonoBehaviour
{
    [SerializeField] private TextMeshPro dpsText; // World-space TMP (not UGUI)

    private readonly List<(float time, float damage)> _hits = new();

    /// <summary>
    /// Called by CharacterProjectile (or whatever collision handler the project uses)
    /// when a player spell hits this dummy.
    /// </summary>
    public void RegisterHit(float damage)
    {
        _hits.Add((Time.time, damage));
    }

    private void Update()
    {
        // Prune hits older than 3 seconds
        float cutoff = Time.time - 3f;
        _hits.RemoveAll(h => h.time < cutoff);

        float dps = _hits.Count > 0 ? _hits.Sum(h => h.damage) / 3f : 0f;
        if (dpsText != null)
            dpsText.text = $"{dps:F1}";
    }
}
```

**Part A — PlaygroundHUDController.cs:**

```csharp
// Assets/Scripts/Playground/PlaygroundHUDController.cs
using UnityEngine;
using UnityEngine.UI;
using TMPro;

public class PlaygroundHUDController : MonoBehaviour
{
    [SerializeField] private ShopUI shopUI;
    [SerializeField] private TextMeshProUGUI tokenCounterText;
    [SerializeField] private Button shopButton;
    [SerializeField] private Button backToMenuButton;

    private void Start()
    {
        shopButton.onClick.AddListener(OpenShop);
        backToMenuButton.onClick.AddListener(BackToMenu);
        RefreshTokenDisplay();
    }

    public void OpenShop()
    {
        if (PlaygroundManager.Instance == null) return;
        shopUI.SetTokenSource(PlaygroundManager.Instance);
        Time.timeScale = 0f;
        shopUI.Show(); // or however existing ShopUI is opened — read ShopUI.cs for correct method name
        RefreshTokenDisplay();
    }

    public void CloseShop()
    {
        shopUI.Hide(suppressSave: true);
        Time.timeScale = 1f;
        RefreshTokenDisplay();
    }

    public void RefreshTokenDisplay()
    {
        if (PlaygroundManager.Instance != null && tokenCounterText != null)
            tokenCounterText.text = $"{PlaygroundManager.Instance.PlaygroundTokens} / 10";
    }

    private void BackToMenu()
    {
        Time.timeScale = 1f; // Ensure not paused when leaving
        SceneLoader.LoadScene("MainMenuScene"); // Use whatever the main menu scene name is — check SceneLoader or existing menu scene name
    }
}
```

Read ShopUI.cs to confirm the correct method for opening the shop panel (Show(), Open(), SetActive(true), etc.) and use that. Also read an existing scene script to confirm the main menu scene name.
